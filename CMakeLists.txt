PROJECT(Launcher)

CMAKE_MINIMUM_REQUIRED(VERSION 2.6)

INCLUDE (CheckIncludeFileCXX)

FIND_PACKAGE(PkgConfig REQUIRED)
PKG_CHECK_MODULES(OGRE REQUIRED OGRE)
PKG_CHECK_MODULES(CEGUI REQUIRED CEGUI)
PKG_CHECK_MODULES(OIS REQUIRED OIS)
PKG_CHECK_MODULES(log4cpp REQUIRED log4cpp)
PKG_CHECK_MODULES(libcurl REQUIRED libcurl)
FIND_PACKAGE(Boost COMPONENTS filesystem thread date_time REQUIRED)

AUX_SOURCE_DIRECTORY(src Launcher_SRCS)
AUX_SOURCE_DIRECTORY(src/Renderers/Ogre Launcher_SRCS)
AUX_SOURCE_DIRECTORY(src/Renderers/CEGUI Launcher_SRCS)                     

INCLUDE_DIRECTORIES(include
                    ${OGRE_INCLUDE_DIRS}
                    ${CEGUI_INCLUDE_DIRS}
                    ${OIS_INCLUDE_DIRS}
                    ${LOG4CPP_INCLUDE_DIRS}
                    ${LIBCURL_INCLUDE_DIRS}
                    ${BOOST_INCLUDE_DIRS})

LINK_DIRECTORIES(${OGRE_LIBRARY_DIRS} ${CEGUI_LIBRARY_DIRS} ${BOOST_LIBRARY_DIRS} ${OIS_LIBRARY_DIRS} ${LIBCURL_LIBRARY_DIRS})
LINK_LIBRARIES(${OGRE_LIBRARIES} ${OIS_LIBRARIES} ${CEGUI_LIBRARIES} boost_thread boost_filesystem boost_date_time curl log4cpp bz2 CEGUIOgreRenderer)

ADD_EXECUTABLE(bin/${PROJECT_NAME} ${Launcher_SRCS})
TARGET_LINK_LIBRARIES(${PROJECT_NAME} ${Launcher_LIBRARIES})
