PROJECT(Launcher)

CMAKE_MINIMUM_REQUIRED(VERSION 2.6)

INCLUDE (CheckIncludeFileCXX)

FIND_PACKAGE(PkgConfig REQUIRED)
PKG_CHECK_MODULES(OGRE REQUIRED OGRE)
PKG_CHECK_MODULES(OIS REQUIRED OIS)
PKG_CHECK_MODULES(log4cpp REQUIRED log4cpp)

AUX_SOURCE_DIRECTORY(src Launcher_SRCS)                   

INCLUDE_DIRECTORIES(include
                    ${OGRE_INCLUDE_DIRS}
                    ${OIS_INCLUDE_DIRS}
                    ${LOG4CPP_INCLUDE_DIRS})

LINK_DIRECTORIES(${OGRE_LIBRARY_DIRS} ${OIS_LIBRARY_DIRS})
LINK_LIBRARIES(${OGRE_LIBRARIES} ${OIS_LIBRARIES} log4cpp bz2)

ADD_EXECUTABLE(bin/${PROJECT_NAME} ${Launcher_SRCS})
TARGET_LINK_LIBRARIES(${PROJECT_NAME} ${Launcher_LIBRARIES})
